                return counter += 1;
            };

        }

    alert(counter());
    alert(counter());
    alert(counter());

        is accessed which is in parent function

        


    Q2
        let count = 0;
    (function () {
        if (count === 0) {
            let count = 1;
            console.log(count); // What is logged?
        }
        console.log(count); // What is logged?
        })();
        

        Output:
            1
            0
        Reason:
            The count variable inside if has block scope that can be accessed only inside {} bracket .The count variable accessed inside function, outisde if statement is accessing 
            global variable count which has global scope.Also this type of function is called  anonymous function
